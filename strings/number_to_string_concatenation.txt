// number to string
console.log(1 + "3"); // = 13

// multiple numbers to string
console.log(1 + 2 + 3 + "4"); // 6 + "4" = 64

// after first string number, the rest is a string
console.log(1 + "2" + 3 + 4); // 1 + "2" + "3 + 4" = 1234


// number to string to multiple numbers
console.log(1 + "2" + (3 + 4 + 5)); // 1 + "2" + 12 = 1212
